\begin{frame}\frametitle{Introdução}
	\begin{itemize}
		\item Demanda crescente por computação.
		\item Em parte atendida pela Lei de Moore.
		\item Avanços tecnológicos tem permitido:
		\begin{itemize}
			\item Aumento exponencial no número de transistores em um circuito integrado;
			\item Redução exponencial no custo de circuito integrado;
			\item Redução no consumo elétrico de sistemas de computação.
		\end{itemize}
	\end{itemize}
\end{frame}

	\end{itemize}
\end{frame}

\begin{frame}\frametitle{Introdução}

		  \end{itemize}
	\end{itemize}
\end{frame}

\section{DTM}

\begin{frame}\frametitle{Memorização dinâmica de traços}
	\begin{itemize}
		\item Técnica que produz ganhos de desempenho em várias aplicações.
		\item Lida muito bem com instruções que possuem dependências.
		\item Permite a execução de múltiplas instruções de uma vez \cite{costa2001explorando}.
	\end{itemize}
\end{frame}

\begin{frame}\frametitle{Memorização dinâmica de traços}
	\begin{itemize}
		\item Instruções redundantes: não possuem efeitos colaterais e já foram executadas anteriormente com os mesmos valores de entrada.
		\item Traço: sequência de instruções redundantes.
	\end{itemize}
\end{frame}

\begin{frame}\frametitle{Memorização dinâmica de traços}
	\begin{itemize}
		\item A memorização dinâmica de traços (DTM) armazena traços e instruções redundantes.
		\item Em execuções posteriores identifica oportunidades de reuso e recupera os resultados anteriores.
	\end{itemize}
\end{frame}

\begin{frame}\frametitle{Memorização dinâmica de traços}
	\begin{itemize}
		\item Os valores utilizados em um traço produzidos externamente a ele compõe o chamado ``contexto de entrada''.
		\item Os valores produzidos em um traço e disponíveis para uso externo a ele compõe o chamado ``contexto de saída''.
		\item Ao reusar um traço, é feito um desvio para a próxima instrução a ser executada.
	\end{itemize}
\end{frame}

\begin{frame}\frametitle{Memorização dinâmica de traços}
	\begin{figure}[!h]
		\label{Fig:ExemploContexto1}
		\caption[Exemplo de pseudocódigo]{
			Exemplo de pseudocódigo.}
		
		\begin{algorithmic}
			\STATE $c \leftarrow c + a$
			\STATE $x \leftarrow c + b$
			\IF{$x \le 5$}
			\STATE $y \leftarrow x * 2$
			\ELSE
			\STATE $x \leftarrow x + 1$
			\ENDIF
		\end{algorithmic}
		\legend{Fonte: elaborada pelo autor}
	\end{figure}
	
	Nesse código, os valores $a, b, c$ compõe o contexto de entrada. 
	
	Os valores $c, x$ e, dependendo do fluxo de execução $y$, compõe o contexto de saída.
\end{frame}

\section{Problema}

\begin{frame}\frametitle{Problema}
	Diversos trabalhos, entre eles \citeonline{costa2001explorando}, \citeonline{pilla2003limits}, \citeonline{silva2006memorizacao} e \citeonline{laurino2007reuso}, testaram a DTM em diferentes formas e todas produziram ganhos de desempenho para diversas aplicações.
\end{frame}

\begin{frame}\frametitle{Problema}
	Além da mudança no desempenho do sistema, incluir a DTM em um processador também pode alterar:
	\begin{itemize}
		\item área de chip;
		\item potência consumida;
		\item latência de ciclo;
	\end{itemize}
\end{frame}

\begin{frame}\frametitle{Problema}
	\begin{itemize}
		\item Trabalhos anteriores que analisavam a DTM utilizaram apenas modelos teóricos ou simulações computacionais.
		\item Impossível ou extremamente difícil de prever tais alterações dessa forma.
	\end{itemize}
\end{frame}

\section{Proposta}

\begin{frame}\frametitle{Proposta}
	Proposta do trabalho:
	\begin{itemize}
		\item Implementar o mecanismo de DTM em uma arquitetura de processador;
		\item Sintetizar o circuito e gravar em uma placa de FPGA;
		\item Executar os programas de \textit{benchmark} do SPEC CPU 2006;
		\item Comparar medições realizadas no processador original com as do processador alterado.
	\end{itemize}
\end{frame}

\begin{frame}\frametitle{Proposta}
	\begin{itemize}
		\item A arquitetura de processador utilizada é o LEON3, da Cobham Gaisler.
		\item O LEON3 implementa a ISA SPARC V8.
		\item A placa de FPGA utilizada é a Altera Cyclone II EP2C35F672C6.
	\end{itemize}
\end{frame}

\begin{frame}\frametitle{Proposta}
	Objetivos do trabalho:
	\begin{itemize}
		\item Definir os \textit{trade-offs} de implementar a DTM;
		\item Saber para quais aplicações práticas a técnica é viável.
	\end{itemize}
\end{frame}
